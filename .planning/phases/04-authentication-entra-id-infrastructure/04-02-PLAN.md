    Imports: Injectable, UnauthorizedException from '@nestjs/common'; JwksClient from 'jwks-rsa';
    jwt (import * as jwt) from 'jsonwebtoken'; AppConfigService from '../config/app-config.service';
    TokenValidator from './token-validator'; CurrentUser from './current-user.type'.

    Class: @Injectable() export class EntraTokenValidator extends TokenValidator.
    Constructor: inject config: AppConfigService. Initialize this.client = new JwksClient({
      cache: true, cacheMaxEntries: 5, cacheMaxAge: 10 * 60 * 1000,
      rateLimit: true, jwksRequestsPerMinute: 5,
      jwksUri: `https://login.microsoftonline.com/${this.config.get('ENTRA_TENANT_ID')}/discovery/v2.0/keys`
    }).

    validate(rawToken: string): Promise<CurrentUser> — four steps:
    Step 1: const decoded = jwt.decode(rawToken, { complete: true }). If !decoded or typeof
    decoded === 'string', throw new UnauthorizedException('AUTH.INVALID_TOKEN_FORMAT').
    Step 2: let publicKey: string. Try: const key = await this.client.getSigningKey(decoded.header.kid);
    publicKey = key.getPublicKey(). Catch: throw new UnauthorizedException('AUTH.KEY_NOT_FOUND').
    Step 3: let payload: Record<string, unknown>. Try: payload = jwt.verify(rawToken, publicKey, {
      issuer: `https://login.microsoftonline.com/${this.config.get('ENTRA_TENANT_ID')}/v2.0`,
      audience: this.config.get('ENTRA_AUDIENCE'),
      algorithms: ['RS256'],      // SECURITY: whitelist only RS256; prevents alg:none confusion attack
    }) as Record<string, unknown>. Catch: throw new UnauthorizedException('AUTH.TOKEN_INVALID').
    Step 4: map claims:
      entraId = payload['oid'] as string | undefined
      CRITICAL — D-03 CORRECTION: email = (payload['preferred_username'] ?? payload['email']) as
      string | undefined. Do NOT use payload['upn'] — upn is absent from Entra v2.0 access tokens.
      tenantId = payload['tid'] as string | undefined
    If any of entraId / email / tenantId is falsy, throw new UnauthorizedException('AUTH.MISSING_REQUIRED_CLAIMS').
    Return { entraId, email, tenantId, displayName: (payload['name'] as string | undefined) ?? null }.

    Do NOT log rawToken, publicKey, or payload at any level — tokens must not appear in logs per
    CLAUDE.md §14 and Phase 3 D-13. Error logging (if any) uses only the error type, not token values.

    FILE 1 — packages/backend/src/auth/jwt-auth.guard.ts:
    Imports: Injectable, CanActivate, ExecutionContext, UnauthorizedException from '@nestjs/common';
    Reflector from '@nestjs/core'; ClsService from 'nestjs-cls'; ExtractJwt from 'passport-jwt';
    IS_PUBLIC_KEY from './decorators/public.decorator'; TokenValidator from './token-validator';
    CurrentUser from './current-user.type'; AppConfigService from '../config/app-config.service'.

    @Injectable() export class JwtAuthGuard implements CanActivate.
    Constructor: tokenValidator: TokenValidator, reflector: Reflector, config: AppConfigService,
    cls: ClsService. All four are injected via NestJS DI; ClsService and AppConfigService are
    globally available (ClsModule.forRoot global:true; AppConfigModule @Global).

    canActivate(context: ExecutionContext): Promise<boolean>:
    1. isPublic = reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [context.getHandler(),
       context.getClass()]). If isPublic, return true (request.user remains null for @GetCurrentUser()).
    2. const request = context.switchToHttp().getRequest<Request & { user?: CurrentUser }>().
    3. Branched token extraction — AUTH_MODE determines the source:
       if (this.config.get('AUTH_MODE') === 'stub'):
         rawToken = request.headers['x-dev-user'] as string | undefined
       else:
         rawToken = ExtractJwt.fromAuthHeaderAsBearerToken()(request) — extracts Bearer token from
         Authorization header (handles malformed headers gracefully; returns null if absent).
    4. if (!rawToken) throw new UnauthorizedException('AUTH.MISSING_TOKEN').
    5. const currentUser = await this.tokenValidator.validate(rawToken).
       Do NOT wrap in try/catch — let UnauthorizedException propagate to GlobalExceptionFilter.
    6. request.user = currentUser.
    7. this.cls.set('user', currentUser) — stores principal for AuthAuditContextProvider and
       any future CLS consumers (e.g., Phase 6 tenant context).
    8. return true.

    NOTE: The guard does NOT catch UnauthorizedException — GlobalExceptionFilter maps all
    HttpException subclasses to the error envelope automatically. Catching here would suppress
    the error code in the response.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-02 | Tampering | `EntraTokenValidator.validate()` — jwt.verify() call | mitigate | `algorithms: ['RS256']` whitelist in jwt.verify options (Task 1) prevents algorithm confusion attack (alg:none, HS256 with public key as secret, etc.). Implemented in EntraTokenValidator Step 3. |
| T-04-03 | Elevation of Privilege | `JwtAuthGuard.canActivate()` — token extraction branch | mitigate | Guard reads X-Dev-User header ONLY when `config.get('AUTH_MODE') === 'stub'`. In entra mode the header is ignored entirely. AUTH_MODE is Zod-validated at startup (Plan 01 T-04-01) so it cannot be overridden at request time. Implemented in Task 2 guard branch logic. |
| T-04-04 | Elevation of Privilege | `EntraTokenValidator.validate()` — jwt.verify() audience and issuer | mitigate | jwt.verify() is called with `audience: ENTRA_AUDIENCE` (client ID GUID, NOT api:// URI per RESEARCH.md aud claim reference) and `issuer: https://login.microsoftonline.com/${ENTRA_TENANT_ID}/v2.0`. Cross-tenant and cross-audience tokens are rejected by jsonwebtoken before claim mapping. Implemented in EntraTokenValidator Step 3. |
| T-04-05 | Information Disclosure | `EntraTokenValidator.validate()` — no token logging | mitigate | rawToken, publicKey, and payload are never passed to any logger. EntraTokenValidator catches errors and re-throws UnauthorizedException with only an error code string (no token content). Consistent with Phase 3 D-13 pinoHttp redaction (Authorization header already redacted). |
