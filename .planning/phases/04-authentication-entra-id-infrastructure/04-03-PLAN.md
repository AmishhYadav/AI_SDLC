EntraTokenValidator.validate(rawToken: string): Promise<CurrentUser>
  - Throws UnauthorizedException('AUTH.INVALID_TOKEN_FORMAT') if jwt.decode fails
  - Throws UnauthorizedException('AUTH.KEY_NOT_FOUND') if JwksClient.getSigningKey throws
  - Throws UnauthorizedException('AUTH.TOKEN_INVALID') if jwt.verify throws
  - Throws UnauthorizedException('AUTH.MISSING_REQUIRED_CLAIMS') if oid/email/tenantId absent
  - Returns CurrentUser: { entraId: payload.oid, email: payload.preferred_username ?? payload.email,
    tenantId: payload.tid, displayName: payload.name ?? null }

StubTokenValidator.validate(rawToken: string): Promise<CurrentUser>
  - Throws UnauthorizedException('AUTH.STUB_MISSING_DEV_USER_HEADER') if rawToken is falsy
  - Returns { entraId: `stub-${rawToken.trim()}`, email: rawToken.trim(),
    tenantId: 'stub-tenant', displayName: null }

JwtAuthGuard.canActivate(context: ExecutionContext): Promise<boolean>
  - Returns true without calling tokenValidator if IS_PUBLIC_KEY metadata is true
  - Throws UnauthorizedException('AUTH.MISSING_TOKEN') if token/header is absent
  - Calls tokenValidator.validate(rawToken); sets request.user and cls.set('user', user)
